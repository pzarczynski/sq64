 94class Game(Chessboard):
 95    """Represents a chess game, including the current board state, move history, and time control."""
 96    fullmoves: int
 97    halfmoves: int
 98    history: list[Transition]
 99    control: tuple[Time, Time]
100
101    def __init__(self, fen: str, control: tuple[Time, Time]) -> None:
102        super().__init__(fen)
103        *_, fn, hc = fen.split()
104        self.fullmoves = int(fn)
105        self.halfmoves = int(hc)
106        self.history = []
107        self.control = control
108
109    def is_insufficient_material(self) -> bool:
110        pieces = [p for _, p in self]
111        if len(pieces) == 2: return True  # only kings
112        return len(pieces) == 3 and any(p.type in (PieceType.BISHOP, PieceType.KNIGHT) for p in pieces)
113
114    def is_stalemate(self) -> bool: return not self.is_check() and not self.legal_moves()
115    def is_checkmate(self) -> bool: return self.is_check() and not self.legal_moves()
116    def is_threefold(self) -> bool: return sum(s.hash == self._hash for s in self.history) >= 2
117    def is_time_over(self) -> bool: return self.control[self.color].is_over()
118    def is_fifty_moves(self) -> bool: return self.halfmoves >= 100
119
120    def is_decisive(self) -> bool:
121        """Returns True if the game has a decisive outcome (checkmate or time over)."""
122        return self.is_checkmate() or self.is_time_over()
123
124    def is_draw(self) -> bool:
125        return (self.is_stalemate() or self.is_fifty_moves() or
126                self.is_threefold() or self.is_insufficient_material())
127
128    def is_game_over(self) -> bool: return self.is_decisive() or self.is_draw()
129
130    @property
131    def outcome(self) -> Outcome | None:
132        if self.is_checkmate(): return Outcome.CHECKMATE
133        if self.is_time_over(): return Outcome.TIME_OVER
134        if self.is_stalemate(): return Outcome.STALEMATE
135        if self.is_threefold(): return Outcome.THREEFOLD
136        if self.is_fifty_moves(): return Outcome.FIFTY_MOVES
137        if self.is_insufficient_material(): return Outcome.INSUFFICIENT
138        return None
139
140    def play(self, move: Move) -> None:
141        """Plays a move on the board, updating the game state and time control accordingly."""
142        self.control[self.color].add_increment()
143        self.history.append(super().push(move))
144        self.fullmoves += self.color ^ 1  # increment after black
145
146        if self[move.to] or self[move.frm].type == PieceType.PAWN:
147            self.halfmoves = 0
148        else:
149            self.halfmoves += 1
150
151    def fen(self) -> str:
152        return super().fen() + f" {self.fullmoves} {self.halfmoves}"
153
154    def copy(self) -> "Game":
155        return Game(self.fen(), (Time(*self.control[0]), Time(*self.control[1])))
156
157    def pgn(self) -> str:
158        """Generates a PGN string representing the game history."""
159        board = self.copy()
160        _ = any(map(board.unpush, reversed(self.history)))
161
162        def pgn_gen() -> Iterator[str]:
163            for i, state in enumerate(self.history):
164                if i == 0:
165                    full = board.fullmoves
166                    yield f"{full}." if board.color else f"{full}..."
167                elif board.color:
168                    yield f"{board.fullmoves}."
169
170                yield f"{board.san(state.move)}" if state.move else "null"
171                _ = board.push(state.move)
172
173            if board.outcome:
174                yield ("0-1" if board.color else "1-0") if board.is_decisive() else "1/2-1/2"
175
176        return " ".join(pgn_gen())

178    def tick(self, dt_ms: int) -> None:
179        """Advances the game clock by the given number of milliseconds."""
180        self.control[self.color].tick(dt_ms)
