 74    def _render(self) -> None:
 75        if not self._text or self._rect.w <= 0 or self._rect.h <= 0:
 76            self._surf = None
 77            return
 78
 79        base_size = 100
 80        base_font = pygame.font.SysFont(None, base_size)

 82
 83        if text_w == 0 or text_h == 0:
 84            self._surf = None
 85            return
 86
 87        ratio = min(self._rect.w / text_w, self._rect.h / text_h)
 88        font_size = max(1, int(base_size * ratio * self._scale))

180class Box(Widget, ABC):
181    """A base class for layout containers that arrange child widgets either horizontally, vertically, or stacked, with optional spacing and weight-based sizing."""
182    def __init__(
183        self,
184        *children: Widget,
185        weights: list[float] | None = None,
186        spacing: float = 0,
187    ) -> None:
188        super().__init__()
189        if weights is not None and len(weights) != len(children):
190            raise ValueError("Length of weights must match number of children")
191
192        self._spacing: float = spacing
193        self._children: list[Widget] = list(children)
194        self._weights: list[float] = weights or ([1.0] * len(children))
195
196    @property
197    def children(self) -> list[Widget]: return self._children
198
199    def handle_event(self, event: Event) -> bool:
200        return any(c.handle_event(event) for c in reversed(self._children))
201
202    def draw(self, screen: pygame.Surface) -> None:
203        for c in self._children:
204            c.draw(screen)
205
206    @abstractmethod
207    def layout(self, rect: pygame.Rect) -> None:
208        super().layout(rect)

210    @classmethod
211    def pad(cls, *ch: Widget, pad: float, spacing: float = 0) -> "Self":
212        w = [1 - 2 * pad] * len(ch)
213        return cls(Widget(), *ch, Widget(), weights=[pad, *w, pad], spacing=spacing)

216class HBox(Box):
217    """A widget that arranges its child widgets horizontally, with optional spacing and weight-based sizing."""
218    def layout(self, rect: pygame.Rect) -> None:
219        super().layout(rect)
220        if not self._children:
221            return
222        spacing = self._spacing * rect.w
223        total_spacing = spacing * (len(self._children) - 1)
224        available_w = rect.w - total_spacing
225        total_weight = sum(self._weights)
226        x = rect.x
227        for c, w in zip(self._children, self._weights, strict=True):
228            w_px = (w / total_weight) * available_w if total_weight > 0 else 0
229            c.layout(pygame.Rect(int(x), rect.y, int(w_px), rect.h))
230            x += w_px + spacing

277class Grid(Widget, Generic[W]):
278    """A widget that arranges its child widgets in a grid layout."""
279    _rows: int
280    _cols: int
281    _spacing: float
282    _bg: Image | None
283    _cells: list[W]
284
285    def __init__(
286        self,
287        factory: Callable[[], W],
288        rows: int = 1,
289        cols: int = 1,
290        spacing: float = 0,
291        bg: Image | None = None,
292    ) -> None:
293        super().__init__()
294        self._rows = rows
295        self._cols = cols
296        self._spacing = spacing
297        self._bg = bg
298        self._cells = [factory() for _ in range(rows * cols)]
299
300    def __setitem__(self, rowcol: tuple[int, int], widget: W) -> None:
301        row, col = rowcol
302        self._cells[row * self._cols + col] = widget
303
304    def __getitem__(self, rowcol: tuple[int, int]) -> W:
305        row, col = rowcol
306        return self._cells[row * self._cols + col]
307
308    def __iter__(self) -> Iterable[W]:
309        yield from self._cells
310
311    def set_bg(self, bg: Image) -> None:
312        self._bg = bg
313
314    def handle_event(self, event: Event) -> bool:
315        return any(c and c.handle_event(event) for c in reversed(self._cells))
316
317    def draw(self, screen: pygame.Surface) -> None:
318        if self._bg:
319            self._bg.draw(screen)
320        for c in self._cells:
321            if c:
322                c.draw(screen)
323
324    def layout(self, rect: pygame.Rect) -> None:
325        super().layout(rect)
326        if self._bg:
327            self._bg.layout(rect)
328
329        cell_w = rect.w / self._cols
330        cell_h = rect.h / self._rows
331        spacing_w = cell_w * self._spacing
332        spacing_h = cell_h * self._spacing
333        for r in range(self._rows):
334            for c in range(self._cols):
335                cell = self._cells[r * self._cols + c]
336                if cell:
337                    x = rect.x + c * cell_w + spacing_w / 2
338                    y = rect.y + r * cell_h + spacing_h / 2
339                    w = cell_w - spacing_w
340                    h = cell_h - spacing_h
341                    cell.layout(pygame.Rect(x, y, w, h))

380class Window(ABC):
381    """A base class for the main application window, managing the main event loop and providing an interface for updating the application state."""
382    screen: pygame.Surface
383    clock: pygame.time.Clock
384    running: bool
385    root: Widget | None
386
387    def __init__(
388        self,
389        width: float,
390        height: float,
391        title: str,
392    ) -> None:
393        self.root = None
394        self.screen = pygame.display.set_mode((width, height), pygame.RESIZABLE)
395        pygame.display.set_caption(title)
396        self.clock = pygame.time.Clock()
397        self.running = False
398
399    def set_root(self, root: Widget) -> None:
400        self.root = root
401
402    def run(self) -> None:
403        if self.root is None:
404            raise RuntimeError("Root widget not set")
405
406        self.running = True
407        self.root.layout(self.screen.get_rect())
408        while self.running:
409            dt = self.clock.tick(60)
410            for event in pygame.event.get():
411                if event.type == pygame.QUIT:
412                    self.running = False
413                elif event.type == pygame.VIDEORESIZE:
414                    self.root.layout(self.screen.get_rect())
415                elif event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
416                    self.root.handle_event(ClickEvent(event.pos))
417            self.update(dt)
418            self.screen.fill((30, 30, 30))
419            self.root.draw(self.screen)
420            pygame.display.flip()
421        pygame.quit()
